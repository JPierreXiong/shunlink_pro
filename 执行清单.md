```typescript
const { searchParams } = new URL(request.url);

// 查询对应计划的站点
const sites = await db()
  .select()
  .from(monitoredSites)
  .innerJoin(user, eq(monitoredSites.userId, user.id))
  .where(
    and(
      eq(monitoredSites.status, 'active'),
      planType === 'free' 
        ? eq(user.planType, 'free')
        : planType === 'paid'
        ? inArray(user.planType, ['base', 'pro'])
        : undefined
    )
  );
```

**任务 1.3**: 添加频率控制
```typescript
function shouldSync(site: any, planType: string): boolean {
  if (!site.lastSyncAt) return true;
  
  const hoursSinceSync = (Date.now() - new Date(site.lastSyncAt).getTime()) / (1000 * 60 * 60);
  
  // 免费用户: 至少间隔 12 小时
  if (planType === 'free') return hoursSinceSync >= 12;
  
  // 付费用户: 至少间隔 3 小时
  if (planType === 'paid') return hoursSinceSync >= 3;
  
  return false;
}

// 应用频率控制
const finalSites = sitesToSync.filter(site => shouldSync(site, planType));
```

### 阶段 2: 创建 QStash 调度（30 分钟）

#### 文件: `scripts/setup-qstash-schedules.ts`

**任务 2.1**: 创建脚本
```typescript
import { Client } from '@upstash/qstash';

const client = new Client({ token: process.env.QSTASH_TOKEN! });
const APP_URL = process.env.NEXT_PUBLIC_APP_URL!;
const CRON_SECRET = process.env.CRON_SECRET!;

async function setupSchedules() {
  // 免费用户调度
  await client.schedules.create({
    destination: `${APP_URL}/api/cron/sync-sites?plan=free`,
    cron: '0 9,21 * * *',  // 每天 2 次
    headers: { 'Authorization': `Bearer ${CRON_SECRET}` },
  });
  
  // 付费用户调度
  await client.schedules.create({
    destination: `${APP_URL}/api/cron/sync-sites?plan=paid`,
    cron: '0 */3 * * *',  // 每天 8 次
    headers: { 'Authorization': `Bearer ${CRON_SECRET}` },
  });
  
  console.log('✅ QStash schedules created');
}

setupSchedules();
```
